# Circle-Area-Calculator
import math

import math

# Define the radius of the circle
radius = 5

# Calculate the area using the formula: Area = pi * r^2
area = math.pi * (radius ** 2)

# Calculate the circumference using the formula: Circumference = 2 * pi * r
circumference = 2 * math.pi * radius

# Print the results
print(f"The radius of the circle is: {radius}")
print(f"The calculated area of the circle is: {area}")
print(f"The calculated circumference of the circle is: {circumference}")
